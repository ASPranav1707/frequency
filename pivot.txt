public class Main
{
	static int Pivot(int[] a) 
	{
	    int li=0,hi=a.length-1;
		while(li<=hi)
		{
			int mid = li+(hi-li)/2;
			if(a[mid]<a[mid-1] && a[mid]<a[mid+1])
			{
				return mid;
			}
			else if(a[li]>a[mid]) 
			{
				hi = mid - 1;
			}
			else if(a[hi]<a[mid]) {
				li = mid + 1;
			}
			
		}
		
		return 0;
	}
	
	public static void main(String[] args) {
		int []a = {8,9,1,2,3,4,5,6,7};
		System.out.println(Pivot(a));
	}

}
